CREATE PROCEDURE spFetchUsers
AS
BEGIN
	SELECT * FROM Users;
END

CREATE PROCEDURE spFetchUser(
@Id int
)
AS
BEGIN
	SELECT * FROM Users WHERE Id = @Id;
END

CREATE PROCEDURE spNewUser(
@Username VARCHAR(MAX),
@Password VARCHAR(MAX),
@Email VARCHAR(MAX),
@Status VARCHAR(MAX),
@Nombre VARCHAR(MAX),
@Apellidos VARCHAR(MAX),
@Roles VARCHAR(MAX)
)
AS
BEGIN
	INSERT INTO Users (Username, Password, Email, Status, Nombre, Apellidos, Roles) VALUES ( @Username, @Password, @Email, @Status, @Nombre, @Apellidos, @Roles);
END

CREATE PROCEDURE spUpdateUser(
@Id int,
@Username VARCHAR(MAX),
@Password VARCHAR(MAX),
@Email VARCHAR(MAX),
@Status VARCHAR(MAX),
@Nombre VARCHAR(MAX),
@Apellidos VARCHAR(MAX),
@Roles VARCHAR(MAX)
)
AS
BEGIN
	UPDATE Users SET Username = @Username, Password = @Password, Email = @Email, Status = @Status, Nombre = @Nombre, Apellidos = @Apellidos, Roles = @Roles WHERE Id = @Id;
END

CREATE PROCEDURE spDeleteUser(
@Id int
)
AS
BEGIN
	DELETE FROM Users WHERE Id = @Id;
END

CREATE PROCEDURE spValidateUser(
@Email VARCHAR (MAX),
@Password VARCHAR (MAX)
)
AS
BEGIN
    SELECT * FROM   Users WHERE  Email = @Email AND Password = @Password;
END
